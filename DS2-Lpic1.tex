\subsection{Customize or Write Simple Scripts}
\begin{lstlisting}[language=bash, caption=Basic Shell Script Template]
#!/bin/bash
# Script: backup.sh
# Description: Backup script example
# Author: Your Name
# Date: $(date)

# Configuration
BACKUP_DIR="/backup"
SOURCE_DIR="/home/user/documents"
DATE=$(date +%Y%m%d_%H%M%S)
LOG_FILE="/var/log/backup.log"

# Functions
log_message() {
    echo "[$(date)] $1" >> "$LOG_FILE"
    echo "$1"
}

# Check if running as root
if [[ $EUID -ne 0 ]]; then
    echo "This script must be run as root" >&2
    exit 1
fi

# Check if source directory exists
if [ ! -d "$SOURCE_DIR" ]; then
    log_message "ERROR: Source directory $SOURCE_DIR not found"
    exit 1
fi

# Create backup directory if it doesn't exist
mkdir -p "$BACKUP_DIR"

# Perform backup
log_message "Starting backup of $SOURCE_DIR"
tar -czf "$BACKUP_DIR/backup_$DATE.tar.gz" "$SOURCE_DIR"

# Check if backup was successful
if [ $? -eq 0 ]; then
    log_message "Backup completed successfully: backup_$DATE.tar.gz"
    # Remove backups older than 30 days
    find "$BACKUP_DIR" -name "*.tar.gz" -mtime +30 -delete
else
    log_message "ERROR: Backup failed"
    exit 1
fi

log_message "Backup process completed"
\end{lstlisting}

\subsection{Basic Network Configuration}
\begin{lstlisting}[language=bash, caption=Network Commands]
# Network interface configuration
ifconfig eth0 192.168.1.100 netmask 255.255.255.0
ip addr add 192.168.1.100/24 dev eth0

# Network interface control
ifconfig eth0 down
ifconfig eth0 up
ip link set eth0 down
ip link set eth0 up

# Routing configuration
route add default gw 192.168.1.1
ip route add default via 192.168.1.1
route -n
ip route show

# DNS configuration
/etc/resolv.conf        # DNS resolver configuration
/etc/hosts              # Local hostname resolution
\end{lstlisting}
